Customer Lifetime Value (LTV)

Finance

Predicts total revenue a customer will generate over their relationship with TripFactory.

// LTV Algorithm - Weighted Historical + Predictive
function calculateLTV(customerId) {
  const customer = getCustomer(customerId);
  
  const avgBookingValue = customer.totalRevenue / customer.bookingCount;
  const purchaseFrequency = customer.bookingCount / 
    monthsSinceFirstBooking(customer);
  const grossMargin = 0.18;
  
  const daysSinceLastBooking = daysSince(customer.lastBooking);
  let retentionRate;
  
  if (daysSinceLastBooking < 90) retentionRate = 0.85;
  else if (daysSinceLastBooking < 180) retentionRate = 0.65;
  else retentionRate = 0.20;
  
  const lifespan = 3 + (customer.loyaltyTier === 'Platinum' ? 2 : 0);
  
  const ltv = (avgBookingValue * purchaseFrequency * grossMargin * lifespan) 
              * retentionRate;
  
  return {
    value: round(ltv, 2),
    tier: assignTier(ltv),
    confidence: calculateConfidence(customer.bookingCount)
  };
}

Vehicle Days in Stock (Capeye)

Operations

Critical metric for inventory management and depreciation calculations.

// Days in Stock with Business Rules
function calculateDaysInStock(vehicle) {
  const today = new Date();
  let startDate;
  
  if (vehicle.dateInStock) {
    startDate = new Date(vehicle.dateInStock);
  } 
  else if (vehicle.createdAt) {
    startDate = new Date(vehicle.createdAt);
  }
  else if (vehicle.regDate) {
    startDate = new Date(vehicle.regDate);
    startDate.setDate(startDate.getDate() - 7);
  }
  else {
    return null;
  }
  
  let days = 0;
  let current = new Date(startDate);
  
  while (current < today) {
    const dayOfWeek = current.getDay();
    if (dayOfWeek !== 0 && dayOfWeek !== 6) {
      days++;
    }
    current.setDate(current.getDate() + 1);
  }
  
  return {
    calendarDays: Math.floor((today - startDate) / (1000*60*60*24)),
    businessDays: days,
    ageingStatus: days > 90 ? 'Critical' : 
                  days > 60 ? 'Warning' : 'Normal'
  };
}
< 60 days: Normal 60-90 days: Warning > 90 days: Critical

Agent Segmentation Score

Sales

Ranks 88,000 agents for targeted marketing and resource allocation.

// Multi-Factor Agent Scoring (0-100)
function calculateAgentScore(agent) {
  const weights = {
    revenue: 0.35,
    volume: 0.25,
    recency: 0.20,
    diversity: 0.15,
    growth: 0.05
  };
  
  const revenueScore = min(agent.monthlyRevenue / 50000, 1);
  const volumeScore = min(agent.monthlyBookings / 100, 1);
  
  const daysSince = daysSinceLastBooking(agent);
  const recencyScore = exp(-daysSince / 30);
  
  const diversityScore = calculateDiversity(
    agent.destinationDistribution
  );
  
  const growthScore = calculateGrowthRate(
    agent.revenueHistory, 6
  );
  
  const finalScore = (
    revenueScore * weights.revenue +
    volumeScore * weights.volume +
    recencyScore * weights.recency +
    diversityScore * weights.diversity +
    growthScore * weights.growth
  ) * 100;
  
  return {
    score: round(finalScore, 1),
    tier: assignAgentTier(finalScore),
    factors: { revenueScore, volumeScore, recencyScore }
  };
}

Dynamic Margin Analysis

Finance

Real-time profitability calculations with market adjustments.

// Margin Calculation with Market Factors
function calculateDynamicMargin(booking) {
  const baseCost = booking.supplierCost;
  const sellingPrice = booking.totalPrice;
  const baseMargin = sellingPrice - baseCost;
  
  const month = getMonth(booking.travelDate);
  const seasonalMultiplier = [
    1.1, 1.05, 1.0, 0.95, 
    0.95, 1.0, 1.15, 1.2, 
    1.1, 1.0, 0.95, 1.05
  ][month];
  
  const volumeDiscount = booking.agent?.monthlyVolume > 50 ? 
    0.02 : 0;
  
  const daysToTravel = daysUntil(booking.travelDate);
  const urgencyPremium = daysToTravel < 7 ? 0.05 : 0;
  
  const adjustedCost = baseCost * (1 - volumeDiscount);
  const adjustedPrice = sellingPrice * 
    (1 + urgencyPremium) * seasonalMultiplier;
  
  const finalMargin = adjustedPrice - adjustedCost;
  const marginPercent = (finalMargin / adjustedPrice) * 100;
  
  return {
    grossMargin: round(finalMargin, 2),
    marginPercent: round(marginPercent, 1),
    health: marginPercent > 20 ? 'Excellent' :
            marginPercent > 15 ? 'Good' :
            marginPercent > 10 ? 'Fair' : 'Poor'
  };
}
